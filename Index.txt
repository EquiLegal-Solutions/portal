    <script>
        // Login Logic
        document.getElementById('loginForm').addEventListener('submit', function(e) {
            e.preventDefault();
            document.getElementById('loginSection').classList.add('hidden');
            document.getElementById('dashboardSection').classList.remove('hidden');
            initChart();
        });

        // Toggle Form
        function toggleForm() {
            document.getElementById('caseModal').classList.toggle('hidden');
        }

        // Chart.js Logic (Investors LOVE charts)
        function initChart() {
            const ctx = document.getElementById('revenueChart').getContext('2d');
            new Chart(ctx, {
                type: 'line',
                data: {
                    labels: ['Jan', 'Feb', 'Mar', 'Apr'],
                    datasets: [{
                        label: 'Revenue Growth',
                        data: [12000, 19000, 15000, 25000],
                        borderColor: '#312e81',
                        tension: 0.4,
                        fill: true,
                        backgroundColor: 'rgba(49, 46, 129, 0.05)'
                    }]
                },
                options: { plugins: { legend: { display: false } }, scales: { y: { display: false }, x: { grid: { display: false } } } }
            });
        }

        // Add Case Logic
        document.getElementById('caseForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const client = document.getElementById('clientName').value;
            const advocate = document.getElementById('advocateName').value;
            const margin = document.getElementById('caseFee').value - document.getElementById('advocateFee').value;
            
            const table = document.getElementById('caseTableBody');
            const row = table.insertRow(0);
            row.className = "hover:bg-indigo-50 transition-colors";
            row.innerHTML = `
                <td class="p-4 font-bold text-slate-700">EJ-NEW-${client.substring(0,3).toUpperCase()}</td>
                <td class="p-4 text-sm text-slate-600 font-medium">${advocate}</td>
                <td class="p-4"><span class="bg-blue-100 text-blue-700 px-2 py-1 rounded-md text-[10px] font-black uppercase">Pending</span></td>
                <td class="p-4 text-right font-mono font-bold text-indigo-900">₹${margin}</td>
            `;
            toggleForm();
            e.target.reset();
        });
    </script>
